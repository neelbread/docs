## Installation

First, install the Bread SDK:

```bash
pip install aibread
```

## Set Up API Key

Set your API key as an environment variable:

```bash
export BREAD_API_KEY='your_api_key_here'
```

<Tip>
Use [python-dotenv](https://pypi.org/project/python-dotenv/) to manage environment variables in a `.env` file and keep credentials out of source control.
</Tip>

## Initialize the Client

<CodeGroup>

```python Sync Client
import os
from aibread import Bread

client = Bread(
    api_key=os.environ.get("BREAD_API_KEY")
)

repos = client.repo.list()
print(repos.items)
```

```python Async Client
import asyncio
from aibread import AsyncBread

async def main():
    async with AsyncBread() as client:
        repos = await client.repo.list()
        print(repos.items)

asyncio.run(main())
```

</CodeGroup>

## Complete Example: From Setup to Rollout

Here's a complete workflow that creates a repository, sets up prompts, configures a target, and runs stim and rollout jobs:

<Steps>
  <Step title="Create Repository & Prompts">
    ```python
    from aibread import Bread
    
    client = Bread()
    
    # Create repository
    repo = client.repo.set(repo_name="my_repo")
    
    # Create system prompt
    client.prompts.set(
        prompt_name="system_prompt",
        repo_name="my_repo",
        messages=[{"role": "system", "content": "You are a coding assistant"}]
    )
    
    # Create user prompt with template variable
    client.prompts.set(
        prompt_name="user_prompt",
        repo_name="my_repo",
        messages=[{"role": "user", "content": "Task: {task}"}]
    )
    ```
  </Step>

  <Step title="Configure Target">
    ```python
    target = client.targets.set(
        target_name="coding_target",
        repo_name="my_repo",
        template="default",
        overrides={
            "generators": [
                {
                    "type": "hardcoded",
                    "numq": 3,
                    "questions": [
                        "Write a function to reverse a string",
                        "Implement binary search",
                        "Create a linked list class"
                    ]
                }
            ],
            "model_name": "Qwen/Qwen3-32B",
            "u": "system_prompt",
            "v": "user_prompt"
        }
    )
    ```
  </Step>

  <Step title="Run Stim Job">
    ```python
    import time
    
    # Start stim job
    client.targets.stim.run(
        target_name="coding_target",
        repo_name="my_repo"
    )
    
    # Poll until complete
    while True:
        status = client.targets.stim.get(
            target_name="coding_target",
            repo_name="my_repo"
        )
        if status.status == "complete":
            print(f"Stim complete! Generated {status.lines} stimuli")
            break
        print(f"Status: {status.status}")
        time.sleep(5)
    ```
  </Step>

  <Step title="Get Stim Output">
    ```python
    output = client.targets.stim.get_output(
        target_name="coding_target",
        repo_name="my_repo",
        limit=100
    )
    
    for stimulus in output.output:
        print(stimulus)
    ```
  </Step>

  <Step title="Run Rollout Job">
    ```python
    # Start rollout job
    client.targets.rollout.run(
        target_name="coding_target",
        repo_name="my_repo"
    )
    
    # Poll until complete
    while True:
        status = client.targets.rollout.get(
            target_name="coding_target",
            repo_name="my_repo"
        )
        if status.status == "complete":
            print(f"Rollout complete! Generated {status.lines} trajectories")
            break
        print(f"Status: {status.status}")
        time.sleep(5)
    ```
  </Step>

  <Step title="Get Rollout Output">
    ```python
    rollout_output = client.targets.rollout.get_output(
        target_name="coding_target",
        repo_name="my_repo",
        limit=100
    )
    
    for trajectory in rollout_output.output:
        print(trajectory)
    ```
  </Step>
</Steps>

## Simple Health Check

Test your connection with a health check:

```python
health = client.health.check()
print(f"Status: {health.status}")  # healthy | degraded
print(f"API Version: {health.version}")
```

## Error Handling

Add error handling to make your code production-ready:

```python
import aibread

try:
    repos = client.repo.list()
except aibread.AuthenticationError:
    print("Invalid API key")
except aibread.APIConnectionError as e:
    print(f"Connection failed: {e}")
except aibread.APIStatusError as e:
    print(f"API error {e.status_code}: {e.response}")
```
